k54.替换数字
```C++
#include <iostream>
#include <string>
using namespace std;

int main() {
    string str;
    cin >> str; // 输入字符串
    string result; // 用于存储结果

    for (size_t i = 0; i < str.size(); i++) {
        if (isdigit(str[i])) { // 判断是否为数字
            result += "number"; // 替换为 "number"
        } else {
            result += str[i]; // 非数字部分直接加入
        }
    }

    cout << result << endl; // 输出结果
    return 0;
}
```
才知道cpp和java的string使用完全不一样，可以随便改变数值，而不是像java一样尽量不变

344.反转字符串
```java
class Solution {
    public void reverseString(char[] s) {
        for(int i=0;i<s.length/2;i++){
            char temp = s[i];
            s[i] = s[s.length-i-1];
            s[s.length-i-1] = temp;
        }
    }
}
```

```java
class Solution {
    public String reverseStr(String s, int k) {
        int chunk = s.length()/(2*k);
        char[] arr = s.toCharArray();
        int start_index = 0;
        while(chunk!=0){
            for(int i=0;i<k/2;i++){
                char front_value = arr[i+start_index];
                char end_value = arr[k-i+start_index-1];
                char temp = front_value;
                arr[i+start_index] = end_value;
                arr[k-i+start_index-1] = temp;
            }
            start_index += 2*k;
            chunk--;
        }

        if(k>s.length()){
            int start = 0;
            int end = s.length()-1;
            while(start<end){
                char temp = arr[start];
                arr[start] = arr[end];
                arr[end] = temp;
                start++;
                end--;
            }
            return new String(arr);
        }

        if(arr.length-start_index < k){
            int start = start_index;
            int end = arr.length-1;
            while(start<end){
                char temp = arr[start];
                arr[start] = arr[end];
                arr[end] = temp;
                start++;
                end--;
            }
        }else{
            int start = start_index;
            int end = start_index+k-1;
            while(start<=end){
                char c1 = arr[start];
                arr[start]=arr[end];
                arr[end] = c1;
                start++;
                end--;
            }
        }
        return new String(arr);
    }
}
```
